Basic Usage

import { chirp } from 'chirpjs';
// Simple success notification 
 chirp.success('Task completed!');
// With custom message 
 chirp.error('Something went wrong', { duration: 5000, position: 'top-right' });

Advanced Options

// Loading with progress 
 const loading = chirp.loading('Processing...');
// Update progress 
 loading.progress(50);
// Complete 
 loading.success('Done!');

Custom Styling
